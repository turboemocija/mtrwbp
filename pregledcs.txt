using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using System.Data;
using System.Data.Sql;


namespace aplikejsn
{
    public partial class Pregled : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            string Path = Server.MapPath("~/biblioteka.xml");
            DataSet ds = new DataSet();
            ds.ReadXml(Path);
            ds.Tables[0].DefaultView.Sort = "citano DESC";
            GridView1.DataSource = ds.Tables[0].DefaultView.ToTable();
            GridView1.DataBind();
        }
    }
}
